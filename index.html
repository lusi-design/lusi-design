} function saveTasks() { localStorage.setItem('tasks', JSON.stringify(tasks)); } function addTask(text) { if (!text.trim()) return; const task = { id: Date.now(), text: text.trim(), completed: false, createdAt: new Date().toISOString() }; tasks.push(task); saveTasks(); renderAll(); document.getElementById('taskInput').value = ''; } function deleteTask(id) { tasks = tasks.filter(task => task.id !== id); saveTasks(); renderAll(); } function toggleTask(id) { const task = tasks.find(t => t.id === id); document.getElementById('pendingTasks').textContent = pending; } function getFilteredTasks() { if (currentFilter === 'active') { return tasks.filter(t => !t.completed); } else if (currentFilter === 'completed') { return tasks.filter(t => t.completed); } return tasks; } function formatDate(dateString) { const date = new Date(dateString); const now = new Date(); const diffDays = Math.floor((now - date) / (1000 * 60 * 60 * 24)); if (diffDays === 0) { return `Today at ${date.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' })}`; } else if (diffDays === 1) { return 'Yesterday'; } else if (diffDays < 7) { return `${diffDays} days ago`; } else { return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' }); } } function escapeHtml(str) { return str.replace(/[&<>]/g, function(m) { if (m === '&') return '&'; if (m === '<') return '<'; if (m === '>') return '>'; return m; }); } function renderTasks() { const filteredTasks = getFilteredTasks(); const taskList = document.getElementById('taskList'); if (filteredTasks.length === 0) { taskList.innerHTML = `

No tasks yet

Add a task to get started
`; return; } taskList.innerHTML = filteredTasks.map((task, index) => `
${escapeHtml(task.text)} ${formatDate(task.createdAt)}
`).join(''); addDragListeners(); } function addDragListeners() { const items = document.querySelectorAll('.task-item'); items.forEach(item => { item.addEventListener('dragstart', handleDragStart); item.addEventListener('dragover', handleDragOver); item.addEventListener('drop', handleDrop); item.addEventListener('dragend', handleDragEnd); }); } function handleDragStart(e) { const item = e.target.closest('.task-item'); if (item) { dragStartIndex = parseInt(item.dataset.index); item.classList.add('dragging'); e.dataTransfer.effectAllowed = 'move'; } } function handleDragOver(e) { e.preventDefault(); e.dataTransfer.dropEffect = 'move'; } function handleDrop(e) { e.preventDefault(); const draggedTask = filteredTasks[dragStartIndex]; if (draggedTask) { const actualTask = tasks.find(t => t.id === draggedTask.id); if (actualTask) { const actualIndex = tasks.findIndex(t => t.id === actualTask.id); const targetActualTask = tasks.find(t => t.id === parseInt(targetItem.dataset.id)); const targetActualIndex = tasks.findIndex(t => t.id === targetActualTask.id); if (actualIndex !== -1 && targetActualIndex !== -1) { tasks.splice(actualIndex, 1); tasks.splice(targetActualIndex, 0, actualTask); saveTasks(); renderAll(); } } } } } } function handleDragEnd(e) { const item = e.target.closest('.task-item'); if (item) { item.classList.remove('dragging'); } dragStartIndex = null; } function renderAll() { updateStats(); renderTasks(); } document.getElementById('addTaskBtn').addEventListener('click', () => { const input = document.getElementById('taskInput'); addTask(input.value); }); document.getElementById('taskInput').addEventListener('keypress', (e) => { if (e.key === 'Enter') { addTask(e.target.value); } }); document.getElementById('clearCompletedBtn').addEventListener('click', () => { clearCompleted(); }); document.querySelectorAll('.filter-tab').forEach(tab => { tab.addEventListener('click', () => { document.querySelectorAll('.filter-tab').forEach(t => t.classList.remove('active')); tab.classList.add('active'); currentFilter = tab.dataset.filter; renderAll(); }); }); loadTasks();
